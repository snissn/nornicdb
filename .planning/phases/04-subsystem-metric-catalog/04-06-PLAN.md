  <name>04-06-04: pkg/replication/peer_metrics_gc.go — D-05b stale-peer GC lifecycle.Component</name>
  <files>pkg/replication/peer_metrics_gc.go, pkg/replication/peer_metrics_gc_test.go</files>
  <behavior>
    Component implementation (~40 LOC):
    ```go
    type peerMetricsGC struct {
        metrics  *observability.ReplicationMetrics
        tracker  *peerTracker  // map[peer]time.Time of last-seen, mutex-protected
        interval time.Duration  // default 5 min, test override 100ms
        staleness time.Duration // default 24h, test override 1s
    }
    func (g *peerMetricsGC) Name() string { return "replication_peer_metrics_gc" }
    func (g *peerMetricsGC) Start(ctx context.Context) error {
        t := time.NewTicker(g.interval)
        defer t.Stop()
        for {
            select {
            case <-ctx.Done(): return nil
            case <-t.C: g.sweep()
            }
        }
    }
    func (g *peerMetricsGC) Shutdown(ctx context.Context) error { return nil }
    func (g *peerMetricsGC) sweep() {
        defer func() { _ = recover() }()
        cutoff := time.Now().Add(-g.staleness)
        for _, peer := range g.tracker.StaleSince(cutoff) {
            // Race-aware: per RESEARCH §Q7 / Pitfall 3, replicator must rebind on reconnect.
            // GC holds tracker lock during DeleteLabelValues to prevent concurrent re-mark.
            g.metrics.LagBytes.DeleteLabelValues(peer)
            g.metrics.LagEntries.DeleteLabelValues(peer)
            g.metrics.RTTSeconds.DeleteLabelValues(peer)
            g.metrics.LastContactSeconds.DeleteLabelValues(peer)
            g.tracker.Forget(peer)
        }
    }
    ```

    `peerTracker` is a tiny mutex-protected map:
    ```go
    type peerTracker struct {
        mu sync.Mutex
        seen map[string]time.Time
    }
    func (t *peerTracker) Mark(peer string)             { ... }
    func (t *peerTracker) StaleSince(cutoff time.Time) []string { ... }
    func (t *peerTracker) Forget(peer string)            { ... }
    ```

    Replicator code marks peers seen (`tracker.Mark(peer)`) at every observation site (heartbeat, replicate, etc.). GC sweep evicts those past the staleness threshold.

    Tests:
